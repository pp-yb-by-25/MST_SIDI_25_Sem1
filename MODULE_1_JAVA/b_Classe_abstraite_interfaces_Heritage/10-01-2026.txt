class Voiture extends Vehicule
{

    // Encapsulation : Dans la classe Voiture, déclarez un attribut privé double prix
    // Utilisez le mot-clé private pour masquer cette donnée à l'extérieur de la classe
    private double prix;


    // Constructeurs : Implémentez un constructeur avec arguments pour Voiture afin d'initialiser ses attributs lors de l'instanciation
    // Rappelez-vous qu'il doit porter le même nom que la classe et n'a aucune valeur de retour
    public Voiture(String matricule, String typeVehicule, double vitesseVehicule , double prix)
    {
        super(matricule,typeVehicule,vitesseVehicule);
        this.prix = prix;
    }

    // Accesseurs (Getters/Setters) : Ajoutez des méthodes publiques pour permettre l'accès et la modification sécurisée du prix
    public double getPrix()
    {
        return this.prix;
    }
    public void setPrix(double prix)
    {
        this.prix = prix;
    }
    // Polymorphisme : Créez une méthode rouler() dans les deux classes. Dans Voiture, redéfinissez-la pour qu'elle ait un comportement spécifique. Cela illustre la capacité du système à choisir la méthode selon le type réel de l'objet

    @Override
    public void rouler()
    {
        System.out.println("La voiture roule à "+this.getVitesseVehicule + " km/h");
    }
    // Destructeur (Finaliseur) : Redéfinissez la méthode finalize() pour afficher "Nettoyage du véhicule" juste avant que l'objet ne soit détruit

    @Override
    public void finalize()
    {
        System.out.println("Nettoyage du vehicule");
    }
    @Override
    public String toString()
    {
        return "Voiture : "+this.getTypeVehicule+" roule à :" + this.vitesseVehicule + " a prix de : " + this.prix;
    }

}
