NomDuProjet/
│
├── app/                            ✅ Module principal de l’application
│   │
│   ├── manifests/
│   │   └── AndroidManifest.xml     # Configuration de l'app (Activities, permissions, etc.)
│   │
│   ├── java/ (ou kotlin/)
│   │   └── com.exemple.monapp/
│   │       ├── MainActivity.java
│   │       └── autres classes      # Activities, Fragments, classes métier, ViewModels
│   │
│   ├── res/                        # Ressources (images, layouts, textes, couleurs…)
│   │   ├── layout/                 # Interfaces XML (activity_main.xml…)
│   │   ├── drawable/               # Images et formes XML
│   │   ├── mipmap/                 # Icônes de l’application
│   │   ├── values/                 # Textes, couleurs, styles, dimensions
│   │   └── menu/                   # Menus (Toolbar, options menu)
│   │
│   └── build.gradle                # Script Gradle spécifique au module app
│
├── gradle/                         # Fichiers et wrapper Gradle
├── build.gradle                     # Script Gradle principal du projet
├── settings.gradle                  # Configuration du projet (modules, chemins)
└── local.properties                 # Configuration locale (SDK path, etc.)
